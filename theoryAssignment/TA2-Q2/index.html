🔍 Search by Title (Text Index)

// Create text index
db.movies.createIndex({ title: "text", plot: "text" })

// Full-text search
db.movies.find({ $text: { $search: "dark knight" } })

🎭 Filter by Genre

db.movies.find({ genre: "Action" })

// Multiple genres (OR)
db.movies.find({ genre: { $in: ["Action", "Sci-Fi"] } })

⭐ Filter by Rating

db.movies.find({ rating: { $gte: 8.0 } })
    .sort({ rating: -1 })
    .limit(20)

📅 Filter by Year Range

db.movies.find({
  year: { $gte: 2010, $lte: 2019 }
}).sort({ year: -1 })

📝 Add a Review (Update)

db.movies.updateOne(
  { _id: ObjectId("...") },
  {
    $push: {
      reviews: {
        userId: ObjectId("..."),
        username: "user123",
        rating: 8.5,
        comment: "Great movie!",
        createdAt: new Date()
      }
    },
    $set: { updatedAt: new Date() }
  }
)

📊 Aggregation: Avg Rating by Genre

db.movies.aggregate([
  { $unwind: "$genre" },
  { $group: {
      _id: "$genre",
      avgRating: { $avg: "$rating" },
      count: { $sum: 1 }
    }
  },
  { $sort: { avgRating: -1 } }
])
